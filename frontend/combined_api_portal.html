// File: backend/server.js require('dotenv').config(); const express = require('express'); const bcrypt = require('bcryptjs'); const jwt = require('jsonwebtoken'); const sqlite3 = require('sqlite3').verbose(); const nodemailer = require('nodemailer'); const cookieParser = require('cookie-parser'); const cors = require('cors'); const crypto = require('crypto'); const app = express(); const PORT = process.env.PORT || 3000; // Production-ready config const JWT_SECRET = process.env.JWT_SECRET || 'REPLACE_WITH_SECURE_JWT_SECRET'; const DB_PATH = process.env.DB_PATH || './data/medorcoin.db'; const SMTP_HOST = process.env.SMTP_HOST; const SMTP_PORT = Number(process.env.SMTP_PORT) || 587; const SMTP_USER = process.env.SMTP_USER; const SMTP_PASS = process.env.SMTP_PASS; const MAIL_FROM = process.env.MAIL_FROM || 'noreply@example.com'; // Initialize SQLite as the production-oriented persistence layer const db = new sqlite3.Database(DB_PATH, (err) => { if (err) { console.error('Failed to open DB:', err); process.exit(1); } else { initDb(); } }); // Simple, production-ready tables function initDb() { db.run( `CREATE TABLE IF NOT EXISTS users ( id INTEGER PRIMARY KEY AUTOINCREMENT, email TEXT UNIQUE NOT NULL, password_hash TEXT NOT NULL, verified INTEGER NOT NULL DEFAULT 0, verification_code TEXT, verification_expires_at INTEGER, created_at INTEGER, updated_at INTEGER )`, (err) => { if (err) console.error(err); } ); db.run( `CREATE TABLE IF NOT EXISTS apis ( id INTEGER PRIMARY KEY AUTOINCREMENT, user_id INTEGER NOT NULL, key TEXT NOT NULL, created_at INTEGER, FOREIGN KEY(user_id) REFERENCES users(id) )`, (err) => { if (err) console.error(err); } ); db.run(`CREATE INDEX IF NOT EXISTS idx_users_email ON users(email)`, (err) => { if (err) console.error(err); }); db.run(`CREATE INDEX IF NOT EXISTS idx_apis_user ON apis(user_id)`, (err) => { if (err) console.error(err); }); } // Email transporter (optional in prod; requires SMTP config) let emailTransporter = null; if (SMTP_HOST && SMTP_USER && SMTP_PASS) { emailTransporter = nodemailer.createTransport({ host: SMTP_HOST, port: SMTP_PORT, secure: SMTP_PORT === 465, auth: { user: SMTP_USER, pass: SMTP_PASS } }); } async function sendVerificationEmail(toEmail, code) { if (!emailTransporter) return; const mailOptions = { from: MAIL_FROM, to: toEmail, subject: 'MedorCoin Verification Code', text: `Your verification code is ${code}`, html: `

Your verification code is ${code}

` }; try { await emailTransporter.sendMail(mailOptions); } catch (err) { console.error('Failed to send verification email:', err); } } // Helpers function generateToken(userId) { return jwt.sign({ userId }, JWT_SECRET, { expiresIn: '2h' }); } function verifyTokenOrNull(token) { try { const payload = jwt.verify(token, JWT_SECRET); return payload.userId; } catch { return null; } } function generateApiKey() { return crypto.randomBytes(16).toString('hex'); } // Middleware: extract token from header or cookie and set req.userId function authMiddleware(req, res, next) { const authHeader = req.headers['authorization']; let token = null; if (authHeader && authHeader.startsWith('Bearer ')) { token = authHeader.slice(7); } else if (req.cookies && req.cookies.token) { token = req.cookies.token; } if (!token) return res.status(401).json({ error: 'Unauthorized' }); const userId = verifyTokenOrNull(token); if (!userId) return res.status(401).json({ error: 'Invalid token' }); req.userId = userId; next(); } // App middlewares app.use(express.json()); app.use(cookieParser()); app.use(cors({ origin: 'http://localhost:8080', credentials: true })); // Routes // Register app.post('/api/auth/register', (req, res) => { const { email, password } = req.body; if (!email || !password) { return res.status(400).json({ error: 'Email and password required' }); } // Check existence db.get('SELECT id FROM users WHERE email = ?', [email], (err, row) => { if (err) { console.error(err); return res.status(500).json({ error: 'Internal server error' }); } if (row) return res.status(400).json({ error: 'User already exists' }); // Hash password bcrypt.hash(password, 12, (err, hash) => { if (err) { console.error(err); return res.status(500).json({ error: 'Internal server error' }); } // Create user const now = Date.now(); const verificationCode = Math.floor(100000 + Math.random() * 900000).toString(); const expiresAt = now + 15 * 60 * 1000; db.run( `INSERT INTO users (email, password_hash, verified, verification_code, verification_expires_at, created_at, updated_at) VALUES (?, ?, 0, ?, ?, ?, ?)`, [email, hash, verificationCode, expiresAt, now, now], function (err) { if (err) { console.error(err); return res.status(500).json({ error: 'Internal server error' }); } const userId = this.lastID; const token = generateToken(userId); res.cookie('token', token, { httpOnly: true, sameSite: 'Strict' }); // Attempt to send verification email (best effort in production) sendVerificationEmail(email, verificationCode); return res.json({ message: 'Registered', token }); } ); }); }); }); // Login app.post('/api/auth/login', (req, res) => { const { email, password } = req.body; if (!email || !password) return res.status(400).json({ error: 'Email and password required' }); db.get('SELECT id, password_hash FROM users WHERE email = ?', [email], (err, row) => { if (err) { console.error(err); return res.status(500).json({ error: 'Internal server error' }); } if (!row) return res.status(401).json({ error: 'Invalid credentials' }); bcrypt.compare(password, row.password_hash, (err, ok) => { if (err) { console.error(err); return res.status(500).json({ error: 'Internal server error' }); } if (!ok) return res.status(401).json({ error: 'Invalid credentials' }); const token = generateToken(row.id); res.cookie('token', token, { httpOnly: true, sameSite: 'Strict' }); res.json({ message: 'Logged in', token }); }); }); }); // Verify email app.post('/api/verify', (req, res) => { const tokenFromCookie = req.cookies.token; const tokenFromHeader = (req.headers['authorization'] || '').startsWith('Bearer ') ? req.headers['authorization'].split(' ')[1] : null; const token = tokenFromCookie || tokenFromHeader; const userId = verifyTokenOrNull(token); if (!token || !userId) return res.status(401).json({ success: false, message: 'No valid token' }); const { code } = req.body; if (!code) return res.status(400).json({ success: false, message: 'Verification code required' }); db.get('SELECT verification_code, verification_expires_at, verified FROM users WHERE id = ?', [userId], (err, row) => { if (err) { console.error(err); return res.status(500).json({ success: false, message: 'Internal server error' }); } if (!row) return res.status(401).json({ success: false, message: 'User not found' }); if (Date.now() > (row.verification_expires_at || 0)) { return res.status(400).json({ success: false, message: 'Verification code expired' }); } if (row.verification_code !== code) { return res.status(400).json({ success: false, message: 'Invalid verification code' }); } db.run('UPDATE users SET verified = 1, verification_code = NULL, verification_expires_at = NULL, updated_at = ? WHERE id = ?', [Date.now(), userId], (err) => { if (err) { console.error(err); return res.status(500).json({ success: false, message: 'Internal server error' }); } return res.json({ success: true, message: 'Email verified' }); }); }); }); // Get user APIs app.get('/api/user/apis', authMiddleware, (req, res) => { const userId = req.userId; db.all('SELECT key FROM apis WHERE user_id = ?', [userId], (err, rows) => { if (err) { console.error(err); return res.status(500).json({ error: 'Internal server error' }); } const keys = rows.map(r => r.key); res.json(keys); }); }); // Generate new API key (requires verification) app.post('/api/apikey/new', authMiddleware, (req, res) => { const userId = req.userId; // Ensure user is verified db.get('SELECT verified FROM users WHERE id = ?', [userId], (err, row) => { if (err) { console.error(err); return res.status(500).json({ error: 'Internal server error' }); } if (!row || !row.verified) { return res.status(403).json({ error: 'Not verified' }); } const apiKey = generateApiKey(); const now = Date.now(); db.run('INSERT INTO apis (user_id, key, created_at) VALUES (?, ?, ?)', [userId, apiKey, now], (err) => { if (err) { console.error(err); return res.status(500).json({ error: 'Internal server error' }); } res.json({ apiKey }); }); }); }); // Optional: health check app.get('/health', (req, res) => res.json({ status: 'ok' })); // Start app.listen(PORT, () => { console.log(`MedorCoin backend running on port ${PORT}`); }); // Graceful shutdown process.on('SIGINT', () => { try { db.close(); } catch (_) {} process.exit(0); }); process.on('SIGTERM', () => { try { db.close(); } catch (_) {} process.exit(0); });
